        {
          h=h%12;
          let hours=(h*360)/12 +(m*360)/(12*60);
          let min =(m*360)/(60);
          let angle =Math.abs(hours-min);
          if (angle>180){
            angle = 360-180;
          }
          return angle;
        };

var Perfect_Check = (N) => 
        {
            sum = 1;
          if(N===1){
            return "YES"
          }
          else if(N%2===1){
            return "NO"
          }
          else{
            
            for(let i=2;i*i<=N;i++){
              if(N%i===0){
                sum += i+N/i
              }
            }
            
            return sum===N?"YES":"NO"
          }
           
                
        };

var Reverse_Number = (N) => 
        {
           rev = 0
          while(N>0){
            r = N%10;
            rev = rev*10+r
            N = parseInt(N/10);
            
          }
        return rev
        
        
        };
